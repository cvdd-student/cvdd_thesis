def solve():
    with open("input.txt", "r") as f:
        lines = f.readlines()
    
    total_points = 0
    cards = []
    for line in lines:
        card_info = line.strip().split(":")
        card_num = int(card_info[0].split()[1])
        winning_numbers_str, your_numbers_str = card_info[1].split("|")
        winning_numbers = set(map(int, winning_numbers_str.strip().split()))
        your_numbers = set(map(int, your_numbers_str.strip().split()))
        
        matches = len(winning_numbers.intersection(your_numbers))
        if matches > 0:
            points = 1
            for i in range(1, matches):
                points *= 2
            total_points += points
        
    return total_points


if __name__ == "__main__":
    result = solve()
    print(result)
